create_table1 <- function(var, var_name) {
  tbl <- table(var)
  pct <- prop.table(tbl) * 100
  data.frame(
    Variable = var_name,

    Frequency = as.vector(tbl),
    Percentage = round(as.vector(pct), 1)
  )
